자바스크립트로 랜덤명언박스만들기

명언이 여기에 나와요

랜덤명언박스
quotes = [
"공부는 배신하지 않는다.",
"포기하지 마라, 끝까지 해보자.",
...
];

softColors = [
"#FFB6B9",
"#FFDAC1",
...
];

-> 공부 명언과 색을 모아놓은 quotes와 softColors라는배열 생성


quoteElement = document.getElementById("quote");

-> id="quote"로 된 곳을 가져와서 그안에 문구를 넣을 준비를 해놓는 코드


function showRandomQuote() {
randomIndex = Math.floor(Math.random() * quotes.length);
const randomColorIndex = Math.floor(Math.random() * softColors.length);
quoteElement.textContent = quotes[randomIndex];
quoteElement.style.color = softColors[randomColorIndex];
}

Math.random() -> 0부터 1 사이의 랜덤 숫자를 만들어줌
Math.floor() -> 소수점을 버리고 정수로 만들어줌
quotes.length -> 문구가 총 몇 개인지 알려줌
randomIndex -> 문구 중 하나를 고르는 랜덤 번호
randomColorIndex -> 색깔 중 하나를 고르는 랜덤 번호


setInterval(showRandomQuote, 5000); -> 5초마다 showRandomQuote을 실행
